[POST]
/register
Headers:
    Content-Type: application/json
Required fields:
    -username (string)
    -email (string)
    -phone (string)
    -password (string)
    -typ ("U" or "I" or "C")
    -address (string)
Returns:
    2XX|4XX: {"message":string}
    
[POST]
/login
Headers:
    Content-Type: application/json
Required fields:
    -email (string)
    -password (string)
Returns:
    4XX:{"message":string}
    2XX:{"token":<jwt>} # string jwt

[DELETE]
/delete
Headers:
    Content-Type: application/json
    Authorization: Bearer <jwt>
Returns:
    4XX:{"message":string}
    2XX:{"message":"account deleted"}
